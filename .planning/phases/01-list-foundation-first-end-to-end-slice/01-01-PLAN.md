const SchemaVersionV1Beta1 = "rollouts.v1beta1"

type Envelope struct {
    SchemaVersion string         `json:"schemaVersion"`
    Kind          string         `json:"kind"`            // "RolloutList" | "Rollout" | "Error"
    Data          interface{}    `json:"data,omitempty"`
    Meta          *EnvelopeMeta  `json:"meta,omitempty"`
    Error         *EnvelopeError `json:"error,omitempty"`
}

type EnvelopeError struct {
    Code       string                 `json:"code"`
    Message    string                 `json:"message"`
    NextAction string                 `json:"nextAction,omitempty"`
    Details    map[string]interface{} `json:"details,omitempty"`
}

type EnvelopeMeta struct {
    FetchedAt        time.Time `json:"fetchedAt,omitempty"`
    UIURL            string    `json:"uiURL,omitempty"`
    Warnings         []string  `json:"warnings,omitempty"`
    AvailableActions []string  `json:"availableActions,omitempty"`
}

// Rollout with the NESTED status block (RESEARCH.md A1) — resolves the kind-collision
// (top-level "kind" = rollout kind; nested "status.kind" = lifecycle bucket).
type Rollout struct {
    ID                  string       `json:"id"`
    FlagKey             string       `json:"flagKey"`
    Kind                string       `json:"kind"`              // "guarded"|"progressive" (rollout kind)
    EnvironmentID       string       `json:"environmentId,omitempty"`
    EnvironmentKey      string       `json:"environmentKey,omitempty"`
    OriginalVariationID string       `json:"originalVariationId"`
    TargetVariationID   string       `json:"targetVariationId"`
    RandomizationUnit   string       `json:"randomizationUnit"`
    RuleIDOrFallthrough string       `json:"ruleIdOrFallthrough"`
    Status              StatusBlock  `json:"status"`            // NESTED — D-02 three-field
    CreatedAt           time.Time    `json:"createdAt"`
    StartedAt           *time.Time   `json:"startedAt,omitempty"`
    EndedAt             *time.Time   `json:"endedAt,omitempty"`
    LatestStageIndex    int          `json:"latestStageIndex"`
    Stages              []Stage      `json:"stages,omitempty"`
    Events              []Event      `json:"events,omitempty"`
    MetricConfigurations []MetricConfiguration `json:"metricConfigurations,omitempty"`
    Links               map[string]Link        `json:"_links,omitempty"`
}

type StatusBlock struct {
    Status string `json:"status"` // raw API enum (e.g. "monitoring_regressed")
    Kind   string `json:"kind"`   // 5-bucket: active|regressed|reverted|paused|completed
    Label  string `json:"label"`  // human-readable with reason inline
}

type RolloutList struct {
    Items []Rollout         `json:"items"`
    Links map[string]Link   `json:"_links,omitempty"`
}
